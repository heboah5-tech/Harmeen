import { useState, useRef, useEffect } from "react";
import { Plus, Minus, Users } from "lucide-react";

const CATEGORIES = [
  { key: "adults",   label: "البالغين",           min: 1 },
  { key: "children", label: "الأطفال",            min: 0 },
  { key: "infants",  label: "الرضع",              min: 0 },
  { key: "special",  label: "الاحتياجات الخاصة", min: 0 },
  { key: "student",  label: "طالب",               min: 0 },
];

export default function PassengerPicker() {
  const [open, setOpen] = useState(false);
  const [counts, setCounts] = useState({ adults: 1, children: 0, infants: 0, special: 0, student: 0 });
  const ref = useRef(null);

  const total = Object.values(counts).reduce((a, b) => a + b, 0);

  const change = (key, delta) => {
    setCounts(prev => {
      const cat = CATEGORIES.find(c => c.key === key);
      const next = Math.max(cat.min, prev[key] + delta);
      return { ...prev, [key]: next };
    });
  };

  useEffect(() => {
    const handler = (e) => { if (ref.current && !ref.current.contains(e.target)) setOpen(false); };
    document.addEventListener("mousedown", handler);
    return () => document.removeEventListener("mousedown", handler);
  }, []);

  return (
    <div ref={ref} className="relative">
      <button
        type="button"
        onClick={() => setOpen(o => !o)}
        className="w-full flex items-center gap-2 bg-muted border border-border rounded-xl py-3 px-4 text-sm text-end text-foreground hover:border-primary focus:outline-none focus:ring-2 focus:ring-primary/30 transition-all"
      >
        <Users className="w-4 h-4 text-muted-foreground flex-shrink-0" />
        <span className="flex-1 text-end">{total} المسافرين</span>
      </button>

      {open && (
        <div className="absolute z-50 mt-1 w-64 bg-background border border-border rounded-2xl shadow-xl p-4 text-end">
          {CATEGORIES.map(cat => (
            <div key={cat.key} className="flex items-center justify-between py-2.5 border-b border-border/50 last:border-0">
              <div className="flex items-center gap-2">
                <button
                  type="button"
                  onClick={() => change(cat.key, 1)}
                  className="w-7 h-7 rounded-full bg-primary/10 text-primary flex items-center justify-center hover:bg-primary hover:text-primary-foreground transition-colors"
                >
                  <Plus className="w-3.5 h-3.5" />
                </button>
                <span className="w-5 text-center font-bold text-foreground text-sm">{counts[cat.key]}</span>
                <button
                  type="button"
                  onClick={() => change(cat.key, -1)}
                  disabled={counts[cat.key] <= cat.min}
                  className="w-7 h-7 rounded-full bg-primary/10 text-primary flex items-center justify-center hover:bg-primary hover:text-primary-foreground transition-colors disabled:opacity-30 disabled:cursor-not-allowed"
                >
                  <Minus className="w-3.5 h-3.5" />
                </button>
              </div>
              <span className="text-sm font-medium text-foreground">{cat.label}</span>
            </div>
          ))}
        </div>
      )}
    </div>
  );
}
